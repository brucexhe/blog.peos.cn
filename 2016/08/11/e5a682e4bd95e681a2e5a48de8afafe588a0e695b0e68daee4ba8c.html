  94. --这时由于我误删数据，我需要将数据恢复到我插入数据之前的状态
  95. --也就是2013-01-11 09:11:15.207这个时间点之前

  99. --这个时候我首先要冷静下来，对事务日志进行一次备份:
  100. USE [master]

  102. TO  DISK = N'E:\MSSQL2008\MSSQL10.MSSQLSERVER\MSSQL\Backup\TestDb.bak' WITH  NO_TRUNCATE , NOFORMAT, NOINIT,
  103. NAME = N'TestDb-事务日志  备份', SKIP, NOREWIND, NOUNLOAD,  NORECOVERY ,
  104. STATS = 10, CHECKSUM
  105. GO
  106. declare @backupSetId as int
  107. select @backupSetId = position
  108. from msdb..backupset
  109. where database_name=N'TestDb'
  110. and backup_set_id=(select max(backup_set_id)
  111. from msdb..backupset where database_name=N'TestDb' )
  112. if @backupSetId is null
  113. begin
  114. raiserror(N'验证失败。找不到数据库“TestDb”的备份信息。', 16, 1)
  115. end
  116. RESTORE VERIFYONLY FROM  DISK = N'E:\MSSQL2008\MSSQL10.MSSQLSERVER\MSSQL\Backup\TestDb.bak' WITH  FILE = @backupSetId,  NOUNLOAD,  NOREWIND
  117. GO
  121. --现在我们来还原数据库,由于我错误的删除了数据   
  124. --首先我根据完整备份进行一次还原,也就是说把数据库还原到
  125. RESTORE DATABASE [TestDb]
  126. FROM  DISK = N'E:\MSSQL2008\MSSQL10.MSSQLSERVER\MSSQL\Backup\TestDb.bak' WITH  FILE = 1,
  127. NORECOVERY,  NOUNLOAD,  STATS = 10
  128. GO
  130. --接下来根据日志备份来恢复数据库:
  132. RESTORE LOG [TestDb] FROM  DISK = N'E:\MSSQL2008\MSSQL10.MSSQLSERVER\MSSQL\Backup\TestDb.bak' WITH  FILE = 2,  NOUNLOAD,
  133. STATS = 10,  STOPAT = N'2013-01-11T09:10:08'--这个时间是我在误删数据之前的时间

  154. 总结:
  155. 为了能够在错误操作或者灾难发生后使数据尽快恢复，数据库就必须要满足一下前提:
  157. 1，数据库的恢复模式必须是完整恢复模式
  158. 2，灾难发生前或者错误删除数据之前必须做过一个完整数据库备份
  159. 3，在上次数据库完整备份之后，如果做过任何日志备份，这些备份现在都能找到
  161. 满足这些要求后，数据库就可以恢复到任何一个时间点了。
  163. 恢复的具体步骤:
  165. 1，灾难发生后备份活动事务日志（也称尾日志）
  166. 2，还原最新完整数据库备份，而不做事务恢复(WITH NORECOVERY)
  167. 3，如果存在差异备份    则还原差异备份，而不做事务恢复(WITH NORECOVERY)
  168. 4，从备份后创建的第一个事务日志开始，使用NORECOVERY一次还原事务日志
  169. 5，恢复数据库到某个时间点（WITH  STOPAT = N'某个时间点'）
